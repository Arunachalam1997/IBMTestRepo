  constructor(props) {
    super(props);
    this.state = {
      NewsSectionData: [],
      NewsSections: [],
      searchQuery: ''
    };
    this.getNewsSection = this.getNewsSection.bind(this);
    this.handleSearchInputChange = this.handleSearchInputChange.bind(this);
  }

  componentDidMount() {
    this.getNewsSection();
  }

  getNewsSection() {
    $.ajax({
      url: this.props.siteURL + "/_api/web/lists/GetByTitle('News Section')/items?$select=Title,Title_Ar,Description_En,Description_Ar,Redirect_Url_En,Redirect_Url_Ar,Image,NewsDate,ID&$orderby=NewsDate desc",
      method: "GET",
      headers: {
        "Accept": "application/json;odata=verbose",
        "If-Match": "*"
      },
      success: (data) => {
        this.setState({ NewsSectionData: data.d.results, NewsSections: data.d.results });
      },
      error: (data) => {
        console.log(JSON.stringify(data));
      }
    });
  }

  handleSearchInputChange(event) {
    const query = event.target.value;
    const filteredNews = this.filterNewsByQuery(query);
    this.setState({ NewsSections: filteredNews, searchQuery: query });
  }



  filterNewsByQuery(query) {
    const { NewsSectionData } = this.state;
    return NewsSectionData.filter((news) =>
      news.Title.toLowerCase().includes(query.toLowerCase()) ||
      news.Description_En.toLowerCase().includes(query.toLowerCase())
    );
  }

  render() {
    const { NewsSections, searchQuery } = this.state;

    return (
      <div>
        <input type="text" placeholder="Search news" value={searchQuery} onChange={this.handleSearchInputChange} />
        <ul>
          {NewsSections.map((news) => (
            <li key={news.ID}>
              <h3>{news.Title}</h3>
              <p>{news.Description_En}</p>
              <img src={news.Image} alt={news.Title} />
            </li>
          ))}
        </ul>
      </div>
    );
  }
}
